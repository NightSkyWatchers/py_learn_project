导入方式1：import 模板文件名 as 别名

{% import '模版/index6.html' as index6 %}

用户名：{{index6.input2('username')}}

密码：{{index6.input2('password',type='password')}}

登录：{{index6.input2('submit',type='submit',value='登录')}}

{% from '模版/index6.html' import input2 %}

用户名：{{input2('username')}}

密码： {{input2('password',type='passwrod')}}

登录： {{ input2('submit',type='submit',value='登录')}}
